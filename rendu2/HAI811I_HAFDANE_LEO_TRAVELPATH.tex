\subsection*{Tâches accomplies jusqu'ici}

\subsubsection*{Authentification utilisateur \& Gestion du profil}
\begin{itemize}
    \item Authentification Firebase : Fonctionnalité de connexion et déconnexion sécurisée.
    \item Profils personnalisables : Les utilisateurs peuvent définir un nom d'utilisateur, une biographie et téléverser une photo de profil (intégrée avec Cloudinary).
    \item Statistiques utilisateur : Suivi en temps réel du nombre de publications, d'abonnés et d'abonnements.
    \item Paramètres : Écran dédié à la gestion des préférences du compte et des configurations de l'application.
\end{itemize}

\subsubsection*{Création de parcours de visite}
\begin{itemize}
    \item Parcours en plusieurs étapes : Possibilité de créer des « TravelPaths » composés de plusieurs étapes séquentielles (monuments, musées, restaurants, loisirs, etc.).
    \item Médias enrichis : Chaque étape prend en charge plusieurs images, automatiquement téléversées sur Cloudinary.
    \item Géolocalisation : Intégration avec Google Maps pour sélectionner et afficher des coordonnées précises pour chaque étape d'un parcours.
    \item Détails de la publication : Champs pour les titres, descriptions et noms de lieux.
\end{itemize}

\subsubsection*{Découverte \& Exploration}
\begin{itemize}
    \item Deux modes d'affichage : Basculer entre une vue Grille/Galerie (axée sur le visuel) et une vue Carte (axée sur la géographie) pour explorer les parcours.
    \item Fonctionnalité de recherche : Une barre supérieure dédiée à la recherche et à la découverte de nouveaux parcours de visite.
    \item Détails interactifs : Un panneau de détails permettant aux utilisateurs d'explorer en profondeur les étapes et images d'un parcours sans quitter l'écran principal.
\end{itemize}

\subsubsection*{Organisation sociale \& personnelle}
\begin{itemize}
    \item Système de favoris : Une section dédiée pour sauvegarder et accéder rapidement aux parcours de visite préférés.
    \item Galerie personnelle : Une galerie propre au profil affichant uniquement les publications de l'utilisateur.
    \item Interactions sociales : Fenêtres de dialogue pour consulter les informations détaillées d'un utilisateur et gérer les connexions sociales.
    \item Système de likes : Possibilité d'aimer ou de retirer un like sur un parcours.
\end{itemize}

\subsubsection*{Infrastructure technique}
\begin{itemize}
    \item Intégration Firestore : Synchronisation en temps réel des publications, profils et étapes.
    \item Intégration Cloudinary : Gestion robuste du téléversement et de l'hébergement des images.
    \item Design Material 3 : Interface moderne et adaptative utilisant NavigationSuiteScaffold pour une navigation fluide sur différentes tailles d'écran.
\end{itemize}

\subsection*{Tâches restantes}

\subsubsection*{Saisie des préférences utilisateur}
\begin{itemize}
    \item Écran dédié permettant à l'utilisateur de renseigner ses critères avant la génération d'un parcours.
    \item Sélection des types d'activités souhaitées : restauration, loisirs, découverte, culture.
    \item Saisie d'une liste de lieux favoris ou obligatoires à inclure dans le parcours.
    \item Définition d'un budget maximum, d'une durée disponible et d'un niveau d'effort (personnes âgées, enfants, personnes malades, etc.).
    \item Renseignement des tolérances météorologiques : sensibilité au froid, à la chaleur, à l'humidité.
\end{itemize}

\subsubsection*{Génération intelligente de parcours}
\begin{itemize}
    \item Calcul automatique d'au moins 2 à 3 options de parcours optimisés selon les préférences saisies.
    \item Proposition de profils distincts : « Économique », « Équilibré » et « Confort ».
    \item Prise en compte des créneaux d'ouverture des lieux (horaires) lors de la génération.
    \item Intégration des données météo du jour pour adapter les suggestions en temps réel.
    \item Possibilité de régénérer un parcours avec des ajustements sans repartir de zéro.
\end{itemize}

\subsubsection*{Présentation riche des parcours générés}
\begin{itemize}
    \item Affichage de l'itinéraire complet avec l'ordre des visites, les distances et les temps de trajet entre chaque étape.
    \item Suggestions de créneaux horaires pour chaque étape (matin, après-midi, soir).
    \item Carte interactive intégrant toutes les étapes du parcours.
    \item Galerie de photos et vidéos associée à chaque étape.
    \item Résumé global du parcours avec métriques claires : coût total estimé, durée totale, niveau d'effort.
\end{itemize}

\subsubsection*{Sauvegarde, partage \& export}
\begin{itemize}
    \item Sauvegarde d'un parcours généré dans les favoris ou dans le profil de l'utilisateur.
    \item Partage d'un parcours avec d'autres utilisateurs depuis l'application.
    \item Export du parcours complet en fichier PDF (étapes, carte, photos, métriques).
\end{itemize}

\subsubsection*{Mode hors-ligne léger}
\begin{itemize}
    \item Mise en cache des données essentielles pour permettre la consultation des parcours sans connexion.
    \item Compression et stockage local des médias associés aux étapes sauvegardées.
\end{itemize}
